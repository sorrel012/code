//Comparator<T>
	- 정렬할 때 비교를 담당하는 compare 메소드를 구현해야 함.
	
	1. 오름차순
		- o1 크면 양수 반환
		- o2 크면 음수 반환
		- 같으면 0 반환
		
		~ public int compare(T o1, T o2) { 
			return o1 - o2;
		  }
		  
		  
	2. 내림차순
		- o1 크면 음수 반환
		- o2 크면 양수 반환
		- 같으면 0 반환		  
		  
		~ public int compare(T o1, T o2) { 
			return o2 - o1;  > 내림차순
		  }
