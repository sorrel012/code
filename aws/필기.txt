// JDK 설치

     1. $ sudo apt-get install openjdk-11-jdk
     
     2. JAVA_HOME 설정
           - $ sudo vim ~/.bashrc

           - i 눌러서 insert모드로 들어간 후  아래 export 붙여넣고 esc

             	export JAVA_HOME=/usr/lib/jvm/java-11-openjdk-amd64
	export PATH=$PATH:$JAVA_HOME/bin

          - $ java --version / $ javac --version / $ echo $JAVA_HOME 으로 확인

     3. 톰캣 설치
          - $ sudo useradd -m -U -d /opt/tomcat -s /bin/false tomcat 로  톰캣 관련 사전 설정

          - $ cd tmp로 이동한 후 $ wget https://dlcdn.apache.org/tomcat/tomcat-9/v9.0.76/bin/apache-tomcat-9.0.76.tar.gz 로 톰캣 설치

          - $ tar -xvf apache-tomcat-9.0.76.tar.gz 로 압축 풀기

          - $ sudo mkdir -p /opt/tomcat  (opt > 추가 패키지 설치 폴더)

          - $ sudo mv apache-tomcat-9.0.76 /opt/tomcat

          - $ sudo ln -s /opt/tomcat/apache-tomcat-9.0.67 /opt/tomcat/latest

          - $ sudo chmod 777 /opt/tomcat/apache-tomcat-9.0.67/bin

          - $ sudo ln -s /opt/tomcat/apache-tomcat-9.0.67 /opt/tomcat/latest

          - $ sudo chown -R tomcat: /opt/tomcat

          - $ sudo chmod +x /opt/tomcat/latest/bin/*.sh

          - $ sudo vim /etc/systemd/system/tomcat.service 후 아래 내용 붙여넣기

[Unit]
Description=Tomcat 9.0.76 servlet container
After=network.target

[Service]
Type=forking

User=tomcat
Group=tomcat

Environment="JAVA_HOME=/usr/lib/jvm/java-11-openjdk-amd64"
Environment="JAVA_OPTS=-Djava.security.egd=file:///dev/urandom"

Environment="CATALINA_BASE=/opt/tomcat/apache-tomcat-9.0.76"
Environment="CATALINA_HOME=/opt/tomcat/apache-tomcat-9.0.76"
Environment="CATALINA_PID=/opt/tomcat/apache-tomcat-9.0.76/temp/tomcat.pid"
Environment="CATALINA_OPTS=-Xms512M -Xmx1024M -server -XX:+UseParallelGC"

ExecStart=/opt/tomcat/apache-tomcat-9.0.76/bin/startup.sh
ExecStop=/opt/tomcat/apache-tomcat-9.0.76/bin/shutdown.sh

[Install]
WantedBy=multi-user.target    

          - $ sudo systemctl daemon-reload

          - $ sudo systemctl start tomcat

          - $ sudo systemctl enable tomcat

          - $ sudo systemctl status tomcat

          - $ sudo ufw allow 8080/tcp

          - $ sudo chmod 777 /opt/tomcat/apache-tomcat-9.0.76/webapps/ 

          - 톰캣 재시작 > webapps > 모든 war 압축 해제
	~ $ sudo systemctl restart tomcat
