				System.out.print("연산자: ");
				operator = scan.nextLine();
				
		 > scan.nextLine()을 한 번 더 사용한다. 
			~	System.out.print("숫자: ");
				num = scan.nextInt(););
		 			
				scan.nextLine(); 
				
				System.out.print("연산자: ");
